## Release Vassal ##

	#Release Vassal Sender
	country_event = {
		id = 99
		title = "EVTNAME99"
		desc = "EVTDESC99"

		is_triggered_only = yes

		# Do nothing
		option = {
			name = "EVTOPTA99"
		}

		# Europe
		option = {
			name = "EVTOPTB99"

			any_country = {
				limit = { vassal_of = THIS NOT = { substate_of = THIS } capital_scope = { continent = europe } }
				country_event = 98
			}
		}

		# Asia
		option = {
			name = "EVTOPTC99"

			any_country = {
				limit = { vassal_of = THIS NOT = { substate_of = THIS } capital_scope = { OR = { continent = asia continent = indochina continent = east_indies continent = oceania } } } #Oceania = Near East, no don't ask me why
				country_event = 98
			}
		}

		# Africa
		option = {
			name = "EVTOPTD99"

			any_country = {
				limit = { vassal_of = THIS NOT = { substate_of = THIS } capital_scope = { OR = { continent = africa continent = east_africa continent = central_africa continent = west_africa continent = south_west_africa } } }
				country_event = 98
			}
		}

		# America and Oceania
		option = {
			name = "EVTOPTE99"

			any_country = {
				limit = { vassal_of = THIS NOT = { substate_of = THIS } capital_scope = { OR = { continent = north_america continent = south_america continent = australia_new_zealand continent = polynesia } } }
				country_event = 98
			}
		}
	}

	#Release Vassal Receiver
	country_event = {
		id = 98
		title = "EVTNAME98"
		desc = "EVTDESC98"

		is_triggered_only = yes

		# Ask to be released
		option = {
			name = "EVTOPTA98"

			FROM = { country_event = 97 }

			ai_chance = { factor = 1 }
		}

		# Do nothing
		option = {
			name = "EVTOPTB98"

			ai_chance = { factor = 0 }
		}
	}

	#Release Vassal Choice
	country_event = {
		id = 97
		title = "EVTNAME97"
		desc = "EVTDESC97"

		allow_multiple_instances = yes
		is_triggered_only = yes

		# Release
		option = {
			name = "EVTOPTA97"

			release_vassal = FROM

			ai_chance = { factor = 0 }
		}

		# Do nothing
		option = {
			name = "EVTOPTB97"

			ai_chance = { factor = 1 }
		}
	}
