	#Description: Morant Bay Rebellion
	country_event = {
		id = 997203
		title = "EVTNAME997203"
		desc = "EVTDESC997203"
		fire_only_once = yes
		major = yes

		trigger = {
			has_country_flag = dr_edward_underhill
			tag = JAM
			vassal_of = ENG
			OR = {
				average_militancy = 5
				ai = yes
			}
		}

		mean_time_to_happen = { months = 3 }

		immediate = { set_global_flag = morant_bay_rebellion }

		# Put down the rebellion
		option = {
			name = "EVTOPTA997203"

			war_exhaustion = 10
			prestige = -10
			badboy = 2
			any_pop = {
				limit = { has_pop_culture = afro_caribbean }
				reduce_pop = 0.99
			}

			ai_chance = { factor = 1 }
		}

		# Declare independence
		option = {
			name = "EVTOPTB997203"

			relation = { who = ENG value = -200 }
			war_exhaustion = 10
			prestige = -10
			badboy = 5
			ENG = {
				release_vassal = THIS
				country_event = 997204
			}
			any_pop = {
				limit = { has_pop_culture = british }
				reduce_pop = 0.85
			}
			primary_culture = kreol
			add_accepted_culture = afro_caribbean

			ai_chance = { factor = 0 }
		}
	}
