#Lucky Nation 1
country_event = {

	id = 66699

	trigger = {
		has_global_flag = lucky_nations_on
		civilized = yes
		ai = yes
		total_pops = 230000
		NOT = { is_greater_power = yes }
		NOT = { has_global_flag = lucky_one }
	}

	mean_time_to_happen = { months = 12 }

	fire_only_once = yes

	title = "EVTNAME32420"
	desc = "EVTDESC32420"
	picture = "revolution"

	option = {
		name = "EVTOPTA32420"
		set_global_flag = lucky_one
		add_country_modifier = {
			name = lucky_nation
			duration = -1
		}
	}
}
